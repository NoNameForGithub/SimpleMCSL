@echo off
:: Minecraft Server Launcher - The Server Toolkit rewrite
:: Author [nonameforgithub 2019]

title MCSL v1.0-prealpha [rewrite]
echo Starting up...
:: Check for temporary files if they weren't deleted in the last session
if exist temp.bat del /q temp.bat

:: Variable setting time
set version=1.0
set dev=1

:: Variable checking time
if %dev%==1 title MCSL v1.0-dev [rewrite]
if %dev%==0 title MCSL v1.0 [rewrite]

goto menu
:menu 
echo [L]atest
echo [C]hangelog
echo [I]nstall or Update to the latest version
SET /P s=Select-[
IF %s%==L GOTO SMCSN
IF %s%==I GOTO inst
IF %s%==l GOTO SMCSN
IF %s%==i GOTO inst
:inst
cls
set cud=.
wget https://hub.spigotmc.org/jenkins/job/BuildTools/lastSuccessfulBuild/artifact/target/BuildTools.jar
set bd="C:\Program Files (x86)\Git\bin\bash.exe"
set bd64="C:\Program Files\Git\bin\bash.exe"
%bd% --login -i -c "java -jar ""%currentdir%\BuildTools.jar"" --rev [mcverhere]
%bd64% --login -i -c "java -jar ""%currentdir%\BuildTools.jar"" --rev [mcverhere]
cd %cud%
java -jar Spigot-[mcvh].jar
if exist Spigot-[mcvh].jar (goto menu) else (wget https://cdn.getbukkit.org/spigot/spigot-mcver.jar)
java -jar Spigot-[mcvh].jar
del BuildTools.jar
cls
goto menu
:SMCSN
cls
title Server console
java -jar Spigot_[mcverhere].jar
cls
java -jar Spigot-[mcverhere].jar
cls
goto menu
